clean_innings <- function(score) {
  score <- as.character(score)
  score[score %in% c("DNB", "TDNB", "absent")] <- NA
  score <- gsub("\\*", "", score)  # Remove not-out asterisk
  return(as.numeric(score))
}
not_out_flag <- function(score) {
  score <- as.character(score)
  return(grepl("\\*", score))
}
